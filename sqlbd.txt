CREATE TABLE usuario (
    id_usuario serial PRIMARY KEY,
    ci VARCHAR(15) NOT NULL,
    nombre VARCHAR(50) NOT NULL,
    primer_apellido VARCHAR(50) NOT NULL,
    segundo_apellido VARCHAR(50) NOT NULL,
    fecha_nacimiento DATE NOT NULL
);
